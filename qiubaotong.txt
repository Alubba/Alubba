INSERT into province (
	province_id,
name,
course_amount,
status,
sort_number,
create_time
)
select a.id as province_id ,a.`name` as name,count(*) as course_amount, 1 as status,0 as sort_number,SYSDATE() as create_time
from 
(
select 	 	bsa.tree_id, substring_index(bsa.tree_id,',',1) id, 
					substring_index(bc.tree_path,',',1) name
		 from base_course bc
left join base_sf_area bsa
			on 	bc.area_id = bsa.id
)a 
GROUP BY  a.id



INSERT into city (
	city_id,
name,
course_amount,
status,
sort_number,
create_time
)
select a.city_id city_id ,a.`name` as name,count(*) as course_amount, 1 as status,0 as sort_number,SYSDATE() as create_time
from 
(
select 	 	bsa.tree_id, substring_index(substring_index(bsa.tree_id,',',2),',',-1) city_id, 
					substring_index(substring_index(bc.tree_path,',',2),',',-1) name
		 from base_course bc
left join base_sf_area bsa
			on 	bc.area_id = bsa.id
)a 
GROUP BY  a.city_id
